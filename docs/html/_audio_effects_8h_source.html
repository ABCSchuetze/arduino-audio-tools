126  public:

128  Tremolo(int16_t &duration_ms, float &depth, uint16_t sampleRate) {
129  int32_t rate_count = static_cast<int32_t>(duration_ms) * sampleRate / 1000;
130  rate_count_half = rate_count / 2;
131 
132  // limit value to max 1.0
133  tremolo_depth = depth>1.0 ? 1.0 : depth;
134  signal_depth = 1.0 - depth;
135  }
136 
137  effect_t process(effect_t input) {
138  if (!active()) return input;
139 
140  float tremolo_factor = tremolo_depth / rate_count_half;
141  int32_t out = (signal_depth * input) + (tremolo_factor * count * input);
142 
143  //saw tooth shaped counter
144  count+=inc;
145  if (count>=rate_count_half){
146  inc = -1;
147  } else if (count<=0) {
148  inc = +1;
149  }
150 
151  return clip(out);
152  }
153 
154 
155  protected:
156  int32_t count = 0;
157  int16_t inc = 1;
