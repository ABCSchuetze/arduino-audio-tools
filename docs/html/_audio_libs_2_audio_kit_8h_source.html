61  audio_hal_iface_samples_t toSampleRate() {
62  LOGD(LOG_METHOD);
63  const static int ia[] = {8000, 11025, 16000, 22050,
64  24000, 32000, 44100, 48000};
65  const static audio_hal_iface_samples_t oa[] = {
66  AUDIO_HAL_08K_SAMPLES, AUDIO_HAL_11K_SAMPLES, AUDIO_HAL_16K_SAMPLES,
67  AUDIO_HAL_22K_SAMPLES, AUDIO_HAL_24K_SAMPLES, AUDIO_HAL_32K_SAMPLES,
68  AUDIO_HAL_44K_SAMPLES, AUDIO_HAL_48K_SAMPLES};
69  int diff = 99999;
70  int result = 0;
71  for (int j = 0; j < 8; j++) {
72  if (ia[j] == sample_rate) {
73  LOGD("-> %d",ia[j])
74  return oa[j];
75  } else {
76  int new_diff = abs(oa[j] - sample_rate);
77  if (new_diff < diff) {
78  result = j;
79  diff = new_diff;
80  }
81  }
82  }
83  LOGE("Sample Rate not supported: %d - using %d", sample_rate, ia[result]);
84  return oa[result];
85  }
