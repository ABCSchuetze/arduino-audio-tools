702 virtual void calculateSampleRate(size_t len) {
703 if (lastTimestamp>0u){
704 if (len>0){
705 uint32_t ms = millis()-lastTimestamp;
706 uint16_t bytes_per_sample = cfg.bits_per_sample/8;
707 if (ms==0){
708 // we can get multiple calls for the same ms
709 samples+= len / cfg.channels / bytes_per_sample;
710 } else {
711 if (lastMs>0){
712 uint16_t rate = samples * 1000 / lastMs;
713
714 if (currentRateValue==0){
715 currentRateValue = rate;
716 } else {
717 currentRateValue = (currentRateValue + rate) / 2;
718 LOGI("effective sample rate: %d - last %d samples", currentRateValue, len);
719 if (cfg.adapt_sample_rate){
720 if (abs(currentRateValue-cfg.sample_rate)>200){
721 cfg.sample_rate = currentRateValue;
722 notifyAudioChange();
723 }
724 }
725 }
726 }
727 lastMs = ms;
728 samples = len / cfg.channels / bytes_per_sample;
729 }
730 }
731 } else {
732 lastTimestamp = millis();
733 }

746// relevant only if use_timer == true
747void IRAM_ATTR timerCallback(void* obj){
748 TimerCallbackAudioStream *src = (TimerCallbackAudioStream*)obj;
749 if (src!=nullptr){
750 //LOGD("%s: %s", LOG_METHOD, src->cfg.rx_tx_mode==RX_MODE ? "RX_MODE":"TX_MODE");
751 if (src->cfg.rx_tx_mode==RX_MODE){
752 // input
753 uint16_t available_bytes = src->frameCallback(src->frame, src->frameSize);
754 uint16_t buffer_available = src->buffer->availableToWrite();
755 if (buffer_available<available_bytes){
756 // if buffer is full make space
757 uint16_t to_clear = available_bytes-buffer_available;
758 uint8_t tmp[to_clear];
759 src->buffer->readArray(tmp,to_clear);
760 }
761 if (src->buffer->writeArray(src->frame, available_bytes)!=available_bytes){
762 //LOGE(UNDERFLOW_MSG);
763 assert(false);
764 }
765 } else {
766 // output
767 if (src->buffer!=nullptr && src->frame!=nullptr && src->frameSize>0){
768 uint16_t available_bytes = src->buffer->readArray(src->frame, src->frameSize);
769 if (available_bytes != src->frameCallback(src->frame, available_bytes)){
770 LOGE(UNDERFLOW_MSG);
771 }
772 }
773 }
774 }
