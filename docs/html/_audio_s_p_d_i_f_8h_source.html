142  // process a single frame
143  int processFrame(int16_t left, int16_t right) {
144  if (!i2sOn) return true; // Sink the data
145  int16_t ms[2];
146  uint16_t hi, lo, aux;
147  uint32_t buf[4];
148 
149  ms[0] = left;
150  ms[1] = right;
151 
152  // S/PDIF encoding:
153  // http://www.hardwarebook.info/S/PDIF
154  // Original sources: Teensy Audio Library
155  // https://github.com/PaulStoffregen/Audio/blob/master/output_spdif2.cpp
156  //
157  // Order of bits, before BMC encoding, from the definition of SPDIF format
158  // PPPP AAAA SSSS SSSS SSSS SSSS SSSS VUCP
159  // are sent rearanged as
160  // VUCP PPPP AAAA 0000 SSSS SSSS SSSS SSSS
161  // This requires a bit less shifting as 16 sample bits align and can be
162  // BMC encoded with two table lookups (and at the same time flipped to LSB
163  // first). There is no separate word-clock, so hopefully the receiver won't
164  // notice.
165 
166  uint16_t sample_left = ms[LEFTCHANNEL];
167  // BMC encode and flip left channel bits
168  hi = spdif_bmclookup[(uint8_t)(sample_left >> 8)];
169  lo = spdif_bmclookup[(uint8_t)sample_left];
170  // Low word is inverted depending on first bit of high word
171  lo ^= (~((int16_t)hi) >> 16);
172  buf[0] = ((uint32_t)lo << 16) | hi;
173  // Fixed 4 bits auxillary-audio-databits, the first used as parity
174  // Depending on first bit of low word, invert the bits
175  aux = 0xb333 ^ (((uint32_t)((int16_t)lo)) >> 17);
176  // Send 'B' preamble only for the first frame of data-block
177  if (frame_num == 0) {
178  buf[1] = VUCP_PREAMBLE_B | aux;
179  } else {
180  buf[1] = VUCP_PREAMBLE_M | aux;
181  }
182 
183  uint16_t sample_right = ms[RIGHTCHANNEL];
184  // BMC encode right channel, similar as above
185  hi = spdif_bmclookup[(uint8_t)(sample_right >> 8)];
186  lo = spdif_bmclookup[(uint8_t)sample_right];
187  lo ^= (~((int16_t)hi) >> 16);
188  buf[2] = ((uint32_t)lo << 16) | hi;
189  aux = 0xb333 ^ (((uint32_t)((int16_t)lo)) >> 17);
190  buf[3] = VUCP_PREAMBLE_W | aux;
191 
192  // Assume DMA buffers are multiples of 16 bytes. Either we write all bytes
193  // or none.
194  int len = 8 * cfg.channels;
195  size_t bytes_written = i2s.write((const uint8_t *)&buf, len);
196  // If we didn't write all bytes, return false early and do not increment
197  // frame_num
198  if (bytes_written != len) return 0;
199  // Increment and rotate frame number
200  if (++frame_num > 191) frame_num = 0;
201  return len;
202  }
