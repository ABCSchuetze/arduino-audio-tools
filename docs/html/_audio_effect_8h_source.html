128class Tremolo : public AudioEffect {
129 public:
131 Tremolo(int16_t &duration_ms, float &depth, uint16_t sampleRate) {
132 int32_t rate_count = static_cast<int32_t>(duration_ms) * sampleRate / 1000;
133 rate_count_half = rate_count / 2;
134
135 // limit value to max 1.0
136 tremolo_depth = depth>1.0 ? 1.0 : depth;
137 signal_depth = 1.0 - depth;
138 }
139
140 effect_t process(effect_t input) {
141 if (!active()) return input;
142
143 float tremolo_factor = tremolo_depth / rate_count_half;
144 int32_t out = (signal_depth * input) + (tremolo_factor * count * input);
145
146 //saw tooth shaped counter
147 count+=inc;
148 if (count>=rate_count_half){
149 inc = -1;
150 } else if (count<=0) {
151 inc = +1;
152 }
153
154 return clip(out);
155 }
156
157
158 protected:
159 int32_t count = 0;
160 int16_t inc = 1;
161 int32_t rate_count_half; // number of samples for on raise and fall
162 float tremolo_depth;
163 float signal_depth;
164 float tremolo_factor ;
165
