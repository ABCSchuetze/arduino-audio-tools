198 
199  // limit value to max 100% and calculate factors
200  float tremolo_depth = p_percent > 100 ? 1.0 : 0.01 * p_percent;
201  float signal_depth = (100.0 - p_percent) / 100.0;

203  float tremolo_factor = tremolo_depth / rate_count_half;
204  int32_t out = (signal_depth * input) + (tremolo_factor * count * input);
