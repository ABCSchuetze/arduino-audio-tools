230

239  }

242  // loop for every value of alpha

312  effectsuite_t getSplineOut(effectsuite_t bufferIndex, int freq) {
313  if (freq < 1) {
314  freq = 1;
315  }
316  const int n0 = floor(bufferIndex);
317  const int n1 = (n0 + freq) % sampleRate;
318  const int n2 = (n0 + (2 * freq)) % sampleRate;
319  const effectsuite_t alpha = bufferIndex - n0;
320  const effectsuite_t a = waveTable[n1];
321  const effectsuite_t c = ((3 * (waveTable[n2] - waveTable[n1])) -
322  (3 * (waveTable[n1] - waveTable[n0]))) *
323  .25;
324  const effectsuite_t b = (waveTable[n2] - waveTable[n1]) - ((2 * c)) / 3;
325  const effectsuite_t d = (-c) / 3;
326  return a + (b * alpha) + (c * alpha * alpha) + (d * alpha * alpha * alpha);
327  }

662  bool setChebyICoefficients(effectsuite_t cutFreq, bool shelfType, effectsuite_t ripple) {
663  // NOTE: coefficient buffers must be cleared as are additive in the
664  // following code
665  std::fill(firCoefficients, firCoefficients + 22, 0);
666  std::fill(iirCoefficients, iirCoefficients + 22, 0);
667 
668  effectsuite_t poles = (effectsuite_t)filterOrder - 1;
669  int order = (int)poles;
670 
671  firCoefficients[2] = 1;
672  iirCoefficients[2] = 1;
673 
674  effectsuite_t Es, Vx, Kx;
675  if (ripple != 0) {
676  Es = sqrt(pow(1 / (1 - ripple), 2) - 1);
677  Vx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) + 1));
678  Kx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) - 1));
679  Kx = cosh(Kx);
680  } else {
681  Vx = 1;
682  Kx = 1;
683  }
684 
685  const effectsuite_t T = 2 * tan(.5);
686  const effectsuite_t W = 2 * pi * cutFreq;
687 
688  effectsuite_t K;
689 
690  if (shelfType == 0)
691  {
692  K = sin(.5 - W / 2) / sin(.5 + W / 2);
693  } else
694  {
695 
696  K = -cos(.5 + W / 2) / cos(W / 2 - .5);
697  }
698 
700  for (int i = 0; i < (order / 2); i++) {
702  const effectsuite_t alpha = pi / (2 * poles) + (i - 1) * (pi / poles);
703 
704  effectsuite_t Rp, Ip;
705  if (ripple != 0) {
706  Rp = -cos(alpha) * sinh(Vx) / Kx;
707  Ip = sin(alpha) * cosh(Vx) / Kx;
708  } else {
709  Rp = -cos(alpha);
710  Ip = sin(alpha);
711  }
712 
713  const effectsuite_t M = pow(Rp, 2) + pow(Ip, 2);
714  const effectsuite_t D = 4 - 4 * Rp * T + M * T;
715 
716  const effectsuite_t X0 = (pow(T, 2)) / D;
717  const effectsuite_t X1 = (2 * pow(T, 2)) / D;
718  const effectsuite_t X2 = X0;
719 
720  const effectsuite_t Y1 = (8 - (2 * M * pow(T, 2))) / D;
721  const effectsuite_t Y2 = (-4 - 4 * Rp * T - M * T) / D;
722 
723  // renamed and inverted from original algorithm
724  const effectsuite_t _D1 = 1 / (1 + Y1 * K - Y2 * pow(K, 2));
725 
726  const effectsuite_t _A0 = (X0 - X1 * K + X2 * pow(K, 2)) * _D1;
727  effectsuite_t _A1 = (-2 * X0 * K + X1 + X1 * pow(K, 2) - 2 * X2 * K) * _D1;
728  const effectsuite_t _A2 = (X0 * pow(K, 2) - X1 * K + X2) * _D1;
729 
730  effectsuite_t _B1 = (2 * K + Y1 + Y1 * pow(K, 2) - 2 * Y2 * K) * _D1;
731  const effectsuite_t B2 = (-(pow(K, 2)) - Y1 * K + Y2) * _D1;
732 
733  if (shelfType == 1) {
734  _A1 = -_A1;
735  _B1 = -_B1;
736  }
737 
738  for (int j = 0; j < 22; j++) {
739  firTemp[j] = firCoefficients[j];
740  iirTemp[j] = iirCoefficients[j];
741  }
742  for (int j = 2; j < 22; j++) {
743  firCoefficients[j] =
744  _A0 * firTemp[j] + _A1 * firTemp[j - 1] + _A2 * firTemp[j - 2];
745  iirCoefficients[j] =
746  iirTemp[j] - _B1 * iirTemp[j - 1] - B2 * iirTemp[j - 2];
747  }
748  }
749 
750  iirCoefficients[2] = 0;
751  for (int j = 0; j < filterOrder; j++) {
752  firCoefficients[j] = firCoefficients[j + 2];
753  iirCoefficients[j] = -iirCoefficients[j + 2];
754  }
756  effectsuite_t SA = 0;
757  effectsuite_t SB = 0;
758  if (shelfType == 0) {
759  for (int j = 0; j < filterOrder; j++) {
760  SA += firCoefficients[j];
761  SB += iirCoefficients[j];
762  }
763  } else {
764  for (int j = 0; j < order; j++) {
765  SA += firCoefficients[j] * pow(-1, j);
766  SB += iirCoefficients[j] * pow(-1, j);
767  }
768  }
769 
770  const effectsuite_t gain = SA / (1 - SB);
771  for (int j = 0; j < filterOrder; j++) {
772  firCoefficients[j] /= gain;
773  }
774 
775  return true;
776  }
