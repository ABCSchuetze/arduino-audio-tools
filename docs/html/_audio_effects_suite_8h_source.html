189 bool setInterpTable() {
190 effectsuite_t *polynomial_normaliser = new effectsuite_t[order];
191 if (!polynomial_normaliser) {
192 return false;
193 }
194 std::fill(polynomial_normaliser, polynomial_normaliser + order, 1);
195 effectsuite_t *alphas = new effectsuite_t[res];
196 if (!alphas) {
197 return false;
198 }
199
200 for (int i = 0; i < res; i++) {
201 alphas[i] = (i / float(res)) - 0.5;
202 }
203
204 effectsuite_t *anchors = new effectsuite_t[order];
205
206 if ((order % 2) == 0) {
207 for (int i = 0; i < order; i++) {
208 anchors[i] = -(effectsuite_t(order) - 1) * 0.5 + effectsuite_t(i);
209 std::fill(interpTable[i], interpTable[i] + res, 1);
210 }
211 } else {
212 for (int i = 0; i < order; i++) {
213 anchors[i] = (-(effectsuite_t(order)) * 0.5) + effectsuite_t(i);
214 }
215 }
216
217 // loop for every value of alpha
218 for (int q = 0; q < res; q++) {
219 // loop for sub polynomial
220 for (int j = 0; j < order; j++) {
221 // loop for each point in subpoly
222 for (int m = 0; m < order; m++) {
223 if (m != j) {
224 if (q == 0) {
225 polynomial_normaliser[j] =
226 polynomial_normaliser[j] * (anchors[j] - anchors[m]);
227 }
228 interpTable[j][q] *= (alphas[q] - anchors[m]);
229 }
230 }
231 interpTable[j][q] /= polynomial_normaliser[j];
232 }
233 }
234
235 delete[] polynomial_normaliser;
236 delete[] alphas;
237 delete[] anchors;
238 return true;
239 }

287 effectsuite_t getSplineOut(effectsuite_t bufferIndex, int freq) {
288 if (freq < 1) {
289 freq = 1;
290 }
291 const int n0 = floor(bufferIndex);
292 const int n1 = (n0 + freq) % sampleRate;
293 const int n2 = (n0 + (2 * freq)) % sampleRate;
294 const effectsuite_t alpha = bufferIndex - n0;
295 const effectsuite_t a = waveTable[n1];
296 const effectsuite_t c = ((3 * (waveTable[n2] - waveTable[n1])) -
297 (3 * (waveTable[n1] - waveTable[n0]))) *
298 .25;
299 const effectsuite_t b = (waveTable[n2] - waveTable[n1]) - ((2 * c)) / 3;
300 const effectsuite_t d = (-c) / 3;
301 return a + (b * alpha) + (c * alpha * alpha) + (d * alpha * alpha * alpha);
302 }

644 bool setChebyICoefficients(effectsuite_t cutFreq, bool shelfType, effectsuite_t ripple) {
645 // NOTE: coefficient buffers must be cleared as are additive in the
646 // following code
647 std::fill(firCoefficients, firCoefficients + 22, 0);
648 std::fill(iirCoefficients, iirCoefficients + 22, 0);
649
650 effectsuite_t poles = (effectsuite_t)filterOrder - 1;
651 int order = (int)poles;
652
653 firCoefficients[2] = 1;
654 iirCoefficients[2] = 1;
655
656 effectsuite_t Es, Vx, Kx;
657 if (ripple != 0) {
658 Es = sqrt(pow(1 / (1 - ripple), 2) - 1);
659 Vx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) + 1));
660 Kx = (1 / poles) * log(1 / Es + sqrt(1 / (pow(Es, 2)) - 1));
661 Kx = cosh(Kx);
662 } else {
663 Vx = 1;
664 Kx = 1;
665 }
666
667 const effectsuite_t T = 2 * tan(.5);
668 const effectsuite_t W = 2 * pi * cutFreq;
669
670 effectsuite_t K;
671
672 if (shelfType == 0)
673 {
674 K = sin(.5 - W / 2) / sin(.5 + W / 2);
675 } else
676 {
677
678 K = -cos(.5 + W / 2) / cos(W / 2 - .5);
679 }
680
682 for (int i = 0; i < (order / 2); i++) {
684 const effectsuite_t alpha = pi / (2 * poles) + (i - 1) * (pi / poles);
685
686 effectsuite_t Rp, Ip;
687 if (ripple != 0) {
688 Rp = -cos(alpha) * sinh(Vx) / Kx;
689 Ip = sin(alpha) * cosh(Vx) / Kx;
690 } else {
691 Rp = -cos(alpha);
692 Ip = sin(alpha);
693 }
694
695 const effectsuite_t M = pow(Rp, 2) + pow(Ip, 2);
696 const effectsuite_t D = 4 - 4 * Rp * T + M * T;
697
698 const effectsuite_t X0 = (pow(T, 2)) / D;
699 const effectsuite_t X1 = (2 * pow(T, 2)) / D;
700 const effectsuite_t X2 = X0;
701
702 const effectsuite_t Y1 = (8 - (2 * M * pow(T, 2))) / D;
703 const effectsuite_t Y2 = (-4 - 4 * Rp * T - M * T) / D;
704
705 // renamed and inverted from original algorithm
706 const effectsuite_t _D1 = 1 / (1 + Y1 * K - Y2 * pow(K, 2));
707
708 const effectsuite_t _A0 = (X0 - X1 * K + X2 * pow(K, 2)) * _D1;
709 effectsuite_t _A1 = (-2 * X0 * K + X1 + X1 * pow(K, 2) - 2 * X2 * K) * _D1;
710 const effectsuite_t _A2 = (X0 * pow(K, 2) - X1 * K + X2) * _D1;
711
712 effectsuite_t _B1 = (2 * K + Y1 + Y1 * pow(K, 2) - 2 * Y2 * K) * _D1;
713 const effectsuite_t B2 = (-(pow(K, 2)) - Y1 * K + Y2) * _D1;
714
715 if (shelfType == 1) {
716 _A1 = -_A1;
717 _B1 = -_B1;
718 }
719
720 for (int j = 0; j < 22; j++) {
721 firTemp[j] = firCoefficients[j];
722 iirTemp[j] = iirCoefficients[j];
723 }
724 for (int j = 2; j < 22; j++) {
725 firCoefficients[j] =
726 _A0 * firTemp[j] + _A1 * firTemp[j - 1] + _A2 * firTemp[j - 2];
727 iirCoefficients[j] =
728 iirTemp[j] - _B1 * iirTemp[j - 1] - B2 * iirTemp[j - 2];
729 }
730 }
731
732 iirCoefficients[2] = 0;
733 for (int j = 0; j < filterOrder; j++) {
734 firCoefficients[j] = firCoefficients[j + 2];
735 iirCoefficients[j] = -iirCoefficients[j + 2];
736 }
738 effectsuite_t SA = 0;
739 effectsuite_t SB = 0;
740 if (shelfType == 0) {
741 for (int j = 0; j < filterOrder; j++) {
742 SA += firCoefficients[j];
743 SB += iirCoefficients[j];
744 }
745 } else {
746 for (int j = 0; j < order; j++) {
747 SA += firCoefficients[j] * pow(-1, j);
748 SB += iirCoefficients[j] * pow(-1, j);
749 }
750 }
751
752 const effectsuite_t gain = SA / (1 - SB);
753 for (int j = 0; j < filterOrder; j++) {
754 firCoefficients[j] /= gain;
755 }
756
757 return true;
758 }

802 gain += firCoefficients[j];

877 effectsuite_t rms(effectsuite_t sample) {
878 rmsBuffer[rmsBufferIndex] = sample;
879 effectsuite_t rmsValue = 0;
880 for (int j = 0; j < rmsBufferIndex; j++) {
881 int i = ((rmsBufferIndex - j) + rmsWindowSize) % rmsWindowSize;
882 rmsValue += rmsBuffer[i] * rmsBuffer[i];
883 }
884
885 // printf("samp: %e\tsum: %e\n", sample, rmsValue);
886 rmsValue /= rmsWindowSize;
887 rmsValue = sqrt(rmsValue);
888
889 rmsBufferIndex++;
890 rmsBufferIndex %= rmsWindowSize;
891
892 return rmsValue;
