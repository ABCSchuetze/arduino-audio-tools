581 }
582 }
583 }break;

644 // writes all input samples
645 void write(const uint8_t *dataIn, size_t len, Print &out){
646 this->data = (T*)dataIn;
647 this->size_val = len;
648 // we process all input samples
649 for (size_t j=0;j<size();j++){
650 T value = data[j];
651 // convert to output format
652 switch(p_convert->target_bits_per_sample){
653 case 8: {
654 int8_t out_value8 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
655 byte arr[p_convert->multiply];
656 memset(arr, out_value8, p_convert->multiply);
657 out.write((const uint8_t*)arr, p_convert->multiply);
658 } break;
659 case 16: {
660 int16_t out_value16 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
661 for (int ch=0;ch<p_convert->multiply;ch++){
662 out.write((const uint8_t*)&out_value16, 2);
663 }
664 } break;
665 case 24: {
666 int32_t out_value24 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
667 for (int ch=0;ch<p_convert->multiply;ch++){
668 out.write((const uint8_t*)&out_value24, 3);
669 }
670 } break;
671 case 32: {
672 int32_t out_value32 = (p_convert->factor * p_convert->format_factor * value) + p_convert->offset;
673 for (int ch=0;ch<p_convert->multiply;ch++){
674 out.write((const uint8_t*)&out_value32, 4);
675 }
676 } break;
677 }
678 }

733 size_t result = 0;
734 if (no_conversion){
735 result = p_out->write(data, len);
736 } else if (convert.multiply>0){
737 switch(p_info_in->bits_per_sample) {
738 case 8: {
739 array8.write(data, len, *p_out);
740 result = len;
741 } break;
742 case 16: {
743 array16.write(data, len/2, *p_out);
744 result = len;

746 case 24: {

770 void setup() {
771 if (p_info_out!=nullptr && p_info_in!=nullptr){
772 LOGI(LOG_METHOD);
773 convert.source_bits_per_sample = p_info_in->bits_per_sample;
774 convert.target_bits_per_sample = p_info_out->bits_per_sample;
775 if (convert.source_bits_per_sample != convert.target_bits_per_sample){
776 convert.format_factor = NumberConverter::maxValue(convert.target_bits_per_sample) / NumberConverter::maxValue(convert.source_bits_per_sample);
777 LOGI("FormatConverterStream: bits %d -> %d - factor %f",convert.source_bits_per_sample, convert.target_bits_per_sample, convert.format_factor);
778 } else {
779 convert.format_factor = 1.0;
780 }
781
782 if (p_info_out->bits_per_sample == p_info_in->bits_per_sample
783 && p_info_out->channels == p_info_in->channels ){
784 no_conversion = true;
785 LOGI("FormatConverterStream: no conversion")
786 } else if (p_info_out->channels == p_info_in->channels){
787 // copy 1:1
788 convert.multiply = 1;
789 convert.step = 1;
790 LOGI("FormatConverterStream: no channel conversion")
791 } else if (p_info_in->channels == 1 && p_info_out->channels>1){
792 // generate multiple output channels per frame
793 convert.multiply = p_info_out->channels;
794 convert.step = 1;
795 LOGI("FormatConverterStream: multiply channels by %d", convert.multiply);
796
797 } else if (p_info_out->channels == 1){
798 convert.multiply = 1;
799 convert.step = p_info_in->channels; // just output 1 channel per frame
800 LOGI("FormatConverterStream: no conversion 1 channel")
801 } else {
802 convert.multiply = 0;
803 LOGE("invalid input vs output channels");
804 }
805 }
806 }
