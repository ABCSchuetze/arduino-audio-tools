92 // defines the pwm_config which will be used to drive the pins
93 pwm_config setupPWMConfig() {
94 LOGD(LOG_METHOD);
95 // setup pwm frequency
96 pwm_config pico_pwm_config = pwm_get_default_config();
97 int wrap_value = maxOutputValue(); // amplitude of square wave (pwm values -amplitude to amplitude) for one byte
98 float pwmClockDivider = static_cast<float>(clock_get_hz(clk_sys)) / (audio_config.pwm_frequency * wrap_value);
99 LOGI("->clock speed is %f", static_cast<float>(clock_get_hz(clk_sys)));
100 LOGI("->divider is %f", pwmClockDivider);
101 pwm_config_set_clkdiv(&pico_pwm_config, pwmClockDivider);
102 pwm_config_set_clkdiv_mode(&pico_pwm_config, PWM_DIV_FREE_RUNNING);
103 //pwm_config_set_phase_correct(&pico_pwm_config, false);
104 pwm_config_set_wrap (&pico_pwm_config, wrap_value);
105 return pico_pwm_config;
