81 public:
83  void calculate(FFTArray<NT> &x) override {
84  // DFT
85  unsigned int N = x.size(), k = N, n;
86  NT thetaT = 3.14159265358979323846264338328L / N;
87  std::complex<NT> phiT = std::complex<NT>(cos(thetaT), -sin(thetaT)), T;
88  while (k > 1) {
89  n = k;
90  k >>= 1;
91  phiT = phiT * phiT;
92  T = 1.0L;
93  for (unsigned int l = 0; l < k; l++) {
94  for (unsigned int a = l; a < N; a += n) {
95  unsigned int b = a + k;
96  std::complex<NT> t = x[a] - x[b];
97  x[a] += x[b];
98  x[b] = t * T;
99  }
100  T *= phiT;
101  }
102  }
103  // Decimate
104  unsigned int m = (unsigned int)log2(N);
105  for (unsigned int a = 0; a < N; a++) {
106  unsigned int b = a;
107  // Reverse bits
108  b = (((b & 0xaaaaaaaa) >> 1) | ((b & 0x55555555) << 1));
109  b = (((b & 0xcccccccc) >> 2) | ((b & 0x33333333) << 2));
110  b = (((b & 0xf0f0f0f0) >> 4) | ((b & 0x0f0f0f0f) << 4));
111  b = (((b & 0xff00ff00) >> 8) | ((b & 0x00ff00ff) << 8));
112  b = ((b >> 16) | (b << 16)) >> (32 - m);
113  if (b > a) {
114  std::complex<NT> t = x[a];
115  x[a] = x[b];
116  x[b] = t;
117  }
118  }
119  }
