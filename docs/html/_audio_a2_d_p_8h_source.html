49RingBuffer<uint8_t> a2dp_buffer(A2DP_BUFFER_SIZE*A2DP_BUFFER_COUNT);
50// flag to indicated that we are ready to process data
51volatile bool is_a2dp_active = false;

53// callback used by A2DP to provide the a2dp_source sound data
54int32_t a2dp_stream_source_sound_data(Frame* data, int32_t len) {
55 LOGD("a2dp_stream_source_sound_data: %d", len);
56 size_t result_len = 0;
57 // at first call we start with some empty data
58 if (!is_a2dp_active){
59 LOGI("setting is_a2dp_active to true");
60 // prevent underflow on first call
61 for(int j=0;j<len;j++){
62 a2dp_buffer.write(0);
63 }
64 is_a2dp_active = true;
65 }
66 // the data in the file must be in int16 with 2 channels
67 size_t result_len_bytes = a2dp_buffer.readArray((uint8_t*)data, len*sizeof(Frame));
68
69 // result is in number of frames
70 result_len = result_len_bytes / sizeof(Frame);
71
72 // Log result
73 if (AudioLogger::instance().level()==AudioLogger::Debug){
74 LOGD("a2dp_stream_source_sound_data %d -> %d", len, result_len);
75 } else {
76 if (result_len<len){
77 LOGW("a2dp_stream_source_sound_data underflow: %d -> %d ", len, result_len);
78 }
79 }
80
81 // allow some other task
82 //yield();
83
84 return result_len;
85}

138 void begin(RxTxMode mode, const char* name){
139 this->mode = mode;
140 this->name = name;
141
142 switch (mode){
143 case TX_MODE:
144 LOGI("Starting a2dp_source...");
145 source(); // allocate object
146 a2dp_source->start((char*)name, a2dp_stream_source_sound_data);
147 while(!a2dp_source->isConnected()){
148 yield();
149 }
150 LOGI("a2dp_source is connected...");
151
152 //is_a2dp_active done in callback to prevent a buffer overflow
153 break;
154
155 case RX_MODE:
156 LOGI("Starting a2dp_sink...");
157 sink(); // allocate object
158 a2dp_sink->set_stream_reader(&a2dp_stream_sink_sound_data, false);
159 a2dp_sink->start((char*)name);
160 while(!a2dp_sink->isConnected()){
161 yield();
162 }
163 LOGI("a2dp_sink is connected...");
164 is_a2dp_active = true;
165 break;
166 }
