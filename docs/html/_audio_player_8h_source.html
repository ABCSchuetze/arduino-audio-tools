214 bool isValidAudioFile(AudioFile& file) {
215 char file_name[MAX_FILE_LEN];

224 AudioFile getFile(const char* dirStr, int pos) {
225 AudioFile dir;
226 AudioFile result;
227 if (sd.exists(dirStr)){
228 LOGI("directory: %s", dirStr);
229 } else {
230 LOGE("directory: %s does not exist", dirStr);
231 stop();
232 }
233
234 dir.open(dirStr);
235 size_t count = 0;
236 getFileAtIndex(dir, pos, count, result);
237 result.getName(file_name, MAX_FILE_LEN);
238 LOGD("-> getFile: '%s': %d", file_name, pos);
239 dir.close();
240 return result;
241 }
242
244 void getFileAtIndex(AudioFile dir, size_t pos, size_t& idx, AudioFile& result) {
245 LOGD("%s: %d", LOG_METHOD, idx);
246 AudioFile file;
247 if (idx > pos) return;
248
249 bool found = false;
250 while (!found && file.openNext(&dir, O_RDONLY )) {
251 //if (idx > pos) return;
252
253 // close all files except result
254 char file_name_act[MAX_FILE_LEN];
255 file.getName(file_name_act, MAX_FILE_LEN);
256 LOGD("-> processing: %s with index %d", file_name_act, idx);
257 if (!file.isHidden()) {
258 if (!file.isDir()) {
259 if (isValidAudioFile(file)) {
260 if (idx == pos) {
261 found = true;
262 result = file;
263 result.getName(file_name, MAX_FILE_LEN);
264 LOGI("==> found: '%s' at index %d", file_name, idx);
265 }
266 idx++;
267 }
268 } else {
269 // process subdirectory
270 getFileAtIndex(file, pos, idx, result);
271 }
272
273 if (!Str(file_name_act).equals(file_name)) {
274 file.getName(file_name, MAX_FILE_LEN);
275 file.close();
276 LOGD("-> close: '%s'", file_name);
277 }
278 }
279 }
280 }

296 AudioSourceURL(AbstractURLStream& urlStream, T(&urlArray)[N], const char* mime, int startPos = 0) {
297 LOGD(LOG_METHOD);
298 this->actual_stream = &urlStream;
299 this->mime = mime;

307 virtual void begin() override {

309 this->pos = 0;

311
313 Stream* selectStream(int idx) override {
314 pos = idx;
315 if (pos < 0) {
316 pos = 0;
317 LOGI("url array out of limits: %d -> %d", idx, pos);
318 }
319 if (pos >= max) {
320 pos = max-1;
321 LOGI("url array out of limits: %d -> %d", idx, pos);
322 }
323 LOGI("selectStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
324 if (started) actual_stream->end();
325 actual_stream->begin(urlArray[pos], mime);
326 started = true;
327 return actual_stream;
328 }
329
331 Stream* nextStream(int offset) override {
332 pos += offset;
333 if (pos < 0 || pos >= max) {
334 pos = 0;
335 }
336 LOGI("nextStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
337 return selectStream(pos);
338 }
339
341 Stream* previousStream(int offset) override {
342 pos -= offset;
343 if (pos < 0 || pos >= max) {
344 pos = max-1;
345 }
346 LOGI("previousStream: %d/%d -> %s", pos, max-1, urlArray[pos]);
347 return selectStream(pos);
348 }

351 return pos;
352 }
